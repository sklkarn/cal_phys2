\begin{equation*} \oint \vec{B}\cdot\vec{\,dl} = \oint B\,dl\cos\theta \end{equation*}
\begin{equation*} = B_{x}(top)\Delta x + B_{x}(top)\Delta z \cos90^{o} \end{equation*}
\begin{equation*} + (-B_{x})(bottom)(-\Delta z)\cos90^{o} + (-B_{x})(bottom)(-\Delta x) \end{equation*}
\begin{equation*} + (-B_{x})(bottom)(-\Delta z)\cos90^{o} + (B_{x})(top)(\Delta z)\cos90^{o} \end{equation*}
\begin{equation*} \therefore \quad \oint \vec{B}\cdot\vec{\,dl} = B_{x}(top)\Delta x + B_{x}(bottom)\Delta x = 2 B_{z}\Delta x =\mu_{o}I_{enc} \end{equation*}
