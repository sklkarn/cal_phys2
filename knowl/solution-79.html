Solution 5.5.3.1.

  1. \begin{equation*} \frac{n_{1}}{s_{1}}+\frac{n'_{1}}{s'_{1}} = \frac{n'_{1}-n_{1}}{R_{1}} \end{equation*}
    \begin{equation*} \text{or,}\quad \frac{1}{30}+\frac{1.5}{s'_{1}} = \frac{1.5-1}{10} \Longrightarrow s'_{1} = 90 \,cm \end{equation*}
    \begin{equation*} s'_{1} +s_{2} = D \Longrightarrow 90 +s_{2} =20 \Longrightarrow s_{2} = -70\,cm \end{equation*}
    \begin{equation*} \frac{1}{s_{2}}+\frac{1}{s'_{2}}=\frac{2}{R_{2}} \end{equation*}
    \begin{equation*} \text{or,}\quad \frac{1}{-70}+\frac{1}{s'_{2}}=\frac{2}{10} \Rightarrow s'_{2} =4.67 \,cm \end{equation*}
    \begin{equation*} s'_{2} +s_{3} = D \Longrightarrow 4.67 +s_{3} =20 \Longrightarrow s_{3} = 15.33 \,cm \end{equation*}
    \begin{equation*} \frac{n_{3}}{s_{3}}+\frac{n'_{3}}{s'_{3}} = \frac{n'_{3}-n_{3}}{R_{3}} \end{equation*}
    \begin{equation*} \text{or,}\quad \frac{1.5}{15.33}+\frac{1}{s'_{3}} = \frac{1-1.5}{-10} \Longrightarrow s'_{3} = -20.9 \,cm \end{equation*}
  2. \begin{equation*} m=-\frac{s'_{1}}{s_{1}}\times-\frac{s'_{2}}{s_{2}}\times-\frac{s'_{3}}{s_{3}} \end{equation*}
    \begin{equation*} =\frac{-90}{30}\times\frac{-4.67}{-70}\times\frac{-(-20.9)}{15.33} \Rightarrow m = -0.273 \end{equation*}
  3. \begin{equation*} m=\frac{y'_{3}}{y1} \Rightarrow -0.273 = \frac{y'_{3}}{25}\Rightarrow y'_{3}= -6.82 \end{equation*}
