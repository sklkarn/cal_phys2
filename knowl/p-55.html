Given: \(q = 10.0\times 10^{-9} \,C; \quad d = 2.0 \,m; \) \(k = 9.0\times10^{9} \,N.m^{2}/C^{2}\)
  1. \begin{equation*} \vec{E_{a}}=\vec{E_{1a}}+\vec{E_{2a}} \end{equation*}
    \begin{equation*} \vec{E_{1a}} =k\frac{q}{r^{2}_{1a}}(-\hat{i}),\quad r_{1a} =1\,m; \end{equation*}
    \begin{equation*} \vec{E_{2a}} =k\frac{-q}{r^{2}_{2a}}(\hat{i}),\quad r_{2a} =1\,m \end{equation*}
    \begin{equation*} \therefore\quad \vec{E_{a}}=-180 (N/C) \hat{i} \end{equation*}
  2. \begin{equation*} \vec{E_{b}}=\vec{E_{1b}}+\vec{E_{2b}} \end{equation*}
    \begin{equation*} \vec{E_{1b}} =k\frac{q}{r^{2}_{1b}}\hat{i},\quad r_{1a} =1\,m; \end{equation*}
    \begin{equation*} \vec{E_{2b}} =k\frac{-q}{r^{2}_{2b}}\hat{i},\quad r_{2b} =3\,m \end{equation*}
    \begin{equation*} \therefore\quad \vec{E_{b}}=8 (N/C) \hat{i} \end{equation*}
  3. \begin{equation*} \vec{E_{c}}=\vec{E_{1c}}+\vec{E_{2c}} \end{equation*}
    \begin{equation*} \vec{E_{1c}} =k\frac{q}{r^{2}_{1c}}\left(\frac{-\hat{i}+\hat{j}}{\sqrt{2}}\right), \quad r_{1c} =\sqrt{2}\,m; \end{equation*}
    \begin{equation*} \vec{E_{2c}} =k\frac{-q}{r^{2}_{2c}}\left(\frac{\hat{i}+\hat{j}}{\sqrt{2}}\right), \quad r_{2c} =\sqrt{2}\,m \end{equation*}
    \begin{equation*} \therefore\quad \vec{E_{c}}=-63.6 (N/C) \hat{i} \end{equation*}
  4. \begin{equation*} \vec{E_{d}}=\vec{E_{1d}}+\vec{E_{2d}} \end{equation*}
    \begin{equation*} \vec{E_{1c}} =k\frac{q}{r^{2}_{1d}}\left(\frac{\hat{i}-\hat{j}}{\sqrt{1^{2}+1^{2}}}\right), \quad r_{1d} = \sqrt{2}\,m; \end{equation*}
    \begin{equation*} \vec{E_{2d}} =k\frac{-q}{r^{2}_{2d}}\left(\frac{3\hat{i}-\hat{j}}{\sqrt{3^{2}+1^{2}}}\right), \quad r_{2d} =\sqrt{10}\,m \end{equation*}
    \begin{equation*} \therefore\quad \vec{E_{d}}=23.3 (N/C) \hat{i} -29.0 (N/C) \hat{j} \end{equation*}
    \begin{equation*} = 37.2 (N/C) \left(\frac{23.3\hat{i}-29.0\hat{j}}{37.2}\right) \end{equation*}
  5. \begin{equation*} \vec{p}=qd\hat{i} \end{equation*}
    \begin{equation*} \therefore\quad \vec{p}=2.0\times 10^{-8}(Cm) \hat{i} \end{equation*}
