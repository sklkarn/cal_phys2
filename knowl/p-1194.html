The intensity due to Fraunhoffer double slits diffraction can be obtained by using eqn. (5.8.13).
\begin{equation*} y = A \left[\int\limits_{-\frac{a}{2}}^{\frac{a}{2}} \sin\left(\omega t -kr+kz\sin\theta \right)\,dz + \int\limits_{d-\frac{a}{2}}^{d+\frac{a}{2}} \sin\left(\omega t -kr+kz\sin\theta \right)\,dz \right] \end{equation*}
\begin{equation*} \therefore \quad y = aA \frac{\sin\alpha}{\alpha}\sin(\omega t -kr) + A \left[\frac{-\cos(\omega t -kr+kz\sin\theta)}{k\sin\theta}\right]_{d-\frac{a}{2}}^{d+\frac{a}{2}} \end{equation*}
\begin{equation*} \text{or,} \quad y = \frac{aA\sin\alpha}{\alpha}\sin(\omega t -kr) - \frac{A}{k\sin\theta} \left[ \cos\left(\omega t -kr+kd\sin\theta + \frac{ka\sin\theta}{2}\right)\right. \end{equation*}
\begin{equation*} \left.-\cos\left(\omega t -kr+kd\sin\theta - \frac{ka\sin\theta}{2}\right) \right] \end{equation*}
\begin{equation*} \text{or,} \quad y = \frac{aA\sin\alpha}{\alpha}\sin(\omega t -kr) + \frac{2A}{k\sin\theta} \left[ \sin\left(\omega t -kr+kd\sin\theta\right)\cdot\sin\left(\frac{ka\sin\theta}{2}\right)\right] \end{equation*}
\begin{equation*} \text{or,} \quad y = \frac{aA\sin\alpha}{\alpha}\sin(\omega t -kr) + \frac{Aa\sin\alpha}{\alpha} \sin\left(\omega t -kr+kd\sin\theta\right) \end{equation*}
\(\left( \because \frac{ka\sin\theta}{2}=\alpha \right)\)
