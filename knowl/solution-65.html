  2. \begin{equation*} v = f_{o}\lambda_{o} \Rightarrow f_{o}= \frac{v}{\lambda_{o}} = \frac{v}{2L}= 15.8 \,Hz \end{equation*}
  3. \begin{equation*} f_{3} =3f_{o} \Rightarrow f_{3}= 47.4 \,Hz \end{equation*}
  4. \begin{equation*} f_{6} =6f_{o} =\frac{v}{\lambda_{6}} \Rightarrow\lambda_{6}= \frac{v}{f_{6}} = 0.667 \,m \end{equation*}
