Force experienced by a current wire in an external magnetic field is given by
\begin{equation*} F=IlB \end{equation*}
here, l is length of the wire.
\begin{equation*} B=\frac{\mu_{o}I}{2\pi r} \end{equation*}
r is the seperation between the wires.
\begin{equation*} \therefore\quad F_{1}= \frac{I_{1}l_{1}\mu_{o}I_{2}}{2\pi r} \end{equation*}
\begin{equation} \frac{F_{1}}{l_{1}} = \frac{\mu_{o}I_{1}I_{2}}{2\pi r} \tag{3.6.1} \end{equation}
From figure
\begin{equation*} T\cos\theta=mg = \mu l \,g \end{equation*}
\begin{equation} \text{or,}\quad T=\frac{\mu l g}{\cos\theta} \tag{3.6.2} \end{equation}
\begin{equation*} \text{also,}\quad \sin\theta=\frac{r/2}{L} = \frac{r}{2L} \end{equation*}
\begin{equation} \text{or,}\quad r=2L\sin\theta \tag{3.6.3} \end{equation}
\begin{equation*} \text{also,}\quad F_{1}=F_{2} = T\sin\theta \end{equation*}
Therefore from eqn. (3.6.1), we have -
\begin{equation*} \frac{F_{1}}{l_{1}} =\frac{T\sin\theta}{l_{1}} =\frac{\mu_{o}I_{1}I_{2}}{2\pi r } \end{equation*}
\begin{equation*} \text{or,}\quad \frac{\mu l g \sin\theta}{l\cos\theta}=\frac{\mu_{o}I^{2}}{2\pi 2\sin\theta } \end{equation*}
\([\because I_{1}=I_{2}=I]\)
\begin{equation*} \text{or,}\quad I^{2}= \frac{\mu \,g \tan\theta \sin\theta 4\pi}{\mu_{o}} \end{equation*}
\begin{equation*} = \frac{4\pi\times 0.001\times9.8\times\tan10^{o}\sin10^{o}}{4\pi\times10^{-7}} = 54.77 \,A \end{equation*}
