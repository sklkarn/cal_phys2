\begin{equation} L=\frac{\phi}{\omega}\tag{5.6.16} \end{equation}
\begin{equation*} or, \quad \phi = L\omega \end{equation*}
\begin{equation} \therefore \quad I = \frac{L\omega}{A} \tag{5.6.17} \end{equation}
Let the area of AC be \(a_{1} \text{,}\) where \(a_{1}= a\cos\theta,\) a being the surface area of AB. Therefore solid angle,
\begin{equation*} \omega = \frac{a_{1}}{r^{2}} =\frac{a\cos\theta}{r^{2}} \end{equation*}
\begin{equation} or, \quad I = \frac{La\cos\theta}{ar^{2}} = \frac{L\cos\theta}{r^{2}} \tag{5.6.18} \end{equation}
