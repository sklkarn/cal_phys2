The choice of signs preceding final terms depends on the current flowing through the two coils. Hence by solving
\begin{equation*} E=I_{1}\left(Z_{g}+R_{1}+j\omega L_{1}\right) + j\omega M I_{2} \end{equation*}
\begin{equation*} 0=I_{2}\left(Z_{l}+R_{2}+j\omega L_{2}\right) - j\omega M I_{1} \end{equation*}
we get -
\begin{equation*} I_{1} = \frac{E\left(Z_{l}+R_{2}+j\omega L_{2}\right)}{\left(Z_{g}+R_{1}+j\omega L_{1}\right) \left(Z_{l}+R_{2}+j\omega L_{2}\right)+\omega^{2}M^{2}} \end{equation*}
and
\begin{equation*} I_{2} = \frac{Ej\omega M}{\left(Z_{g}+R_{1}+j\omega L_{1}\right) \left(Z_{l}+R_{2}+j\omega L_{2}\right)+\omega^{2}M^{2}} \end{equation*}
Hence voltage drop at primary coil,
\begin{equation*} V_{1} = E-I_{1}Z_{g} \end{equation*}
and
\begin{equation*} V_{2} = I_{2}Z_{l} \end{equation*}
In an ideal transformer flux due to one coil is completely passes through the other, i.e.,
\begin{equation*} \Phi_{B1} =\Phi_{B2}, \end{equation*}
hence the coupling constant k becomes unity,
\begin{equation*} \therefore \quad M= k\sqrt{L_{1}L_{2}} =\sqrt{L_{1}L_{2}} \end{equation*}
Since \(R_{1}\) and \(R_{2}\) in an ideal transformer is zero, and reactance of the two coils are so large in comparison to the generator and load impedances, we can write
\begin{equation*} Z_{g}+R_{1}+j\omega L_{1} \approx j\omega L_{1} \end{equation*}
and
\begin{equation*} Z_{l}+R_{2}+j\omega L_{2} \approx j\omega L_{2} \end{equation*}
Hence for an ideal transformer
\begin{equation*} \frac{I_{1}}{I_{2}} = \frac{Z_{l}+R_{2}+j\omega L_{2}}{j\omega M} \approx \frac{L_{2}}{M} = \sqrt{\frac{L_{2}}{L_{1}}} \end{equation*}
but \(L \propto N^{2}\)
\begin{equation*} \therefore \quad \frac{I_{1}}{I_{2}}=\frac{N_{2}}{N_{1}} \end{equation*}
\begin{equation*} \text{or,}\quad N_{1}I_{1}= N_{2}I_{2} \end{equation*}
Also
\begin{equation*} \frac{V_{2}}{V_{1}} = \frac{i_{2}Z_{l}}{E-i_{1}Z_{g}} \approx \frac{I_{2}Z_{l}}{E} \end{equation*}
\begin{equation*} \approx \frac{Ej\omega M}{\left(j\omega L_{1}\right) \left(Z_{l}+j\omega L_{2}\right)+\omega^{2}M^{2}}\cdot\frac{Z_{l}}{E} \end{equation*}
\begin{equation*} \approx \frac{j\omega MZ_{l}}{j\omega L_{1}Z_{l}-\omega^{2} L_{1} L_{2} +\omega^{2}M^{2}} \end{equation*}
\begin{equation*} \text{or,}\quad \frac{V_{2}}{V_{1}} \approx \frac{M}{L_{1}} \approx \sqrt{\frac{L_{2}}{L_{1}}} \approx \frac{N_{2}}{N_{1}} \end{equation*}
This equation shows the voltage transformation property of a transformer. Also we have
\begin{equation*} I_{1}V_{1}=I_{2}V_{2} \end{equation*}
i.e., Input power = output power.
