At achromatic condition,
\begin{equation*} \frac{\omega_{1}}{f_{1}} + \frac{\omega_{2}}{f_{2}} =0 \end{equation*}
\begin{equation*} or,\quad \frac{1}{f_{2}} = -\frac{\omega_{1}}{f_{1}\omega_{2}} = - \frac{2}{3f_{1}} \end{equation*}
Also,
\begin{equation*} \frac{1}{F}=\frac{1}{f_{1}}+\frac{1}{f_{2}} \end{equation*}
\begin{equation*} or,\quad \frac{1}{90}=\frac{1}{f_{1}}-\frac{2}{3f_{1}} =\frac{1}{3f_{1}} \end{equation*}
\begin{equation*} \therefore\quad f_{1}=\frac{90}{3}=30 cm;\quad f_{2}= -\frac{3f_{1}}{2} =-45\,cm \end{equation*}
