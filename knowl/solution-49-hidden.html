Given: \(I_{1} = 10 \,A; \quad a= 0.1 \,m; b=0.3\,m;\) \(c=0.5\,m; I_{2} = 5 \,A.\)
\begin{equation*} \frac{\vec{F_{a}}}{c} = \frac{\mu_{o}I_{1}I_{2} }{2\pi a}\hat{i} \end{equation*}
\begin{equation*} \frac{\vec{F_{b}}}{c} = \frac{\mu_{o}I_{1}I_{2} }{2\pi b}[-\hat{i}] \end{equation*}
\begin{equation*} \vec{F_{top}}=-\vec{F_{bottom}} \end{equation*}
\begin{equation*} \vec{F}=\vec{F_{a}}+ \vec{F_{b}}+\vec{F_{top}} + \vec{F_{bottom}} \end{equation*}
\begin{equation*} \vec{F} = \frac{\mu_{o}I_{1}I_{2}c }{2\pi}\left(\frac{1}{a}-\frac{1}{b}\right)\hat{i} \end{equation*}
\begin{equation*} \vec{F} = 3.33\times10^{-5} N \hat{i} \end{equation*}
