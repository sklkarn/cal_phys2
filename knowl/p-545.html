Case III: If \(\lambda^{2} \lt \omega^{2}_{o},\) then \(\beta\) is imaginary. Let us put
\begin{equation*} \beta = \sqrt{(\lambda^{2}-\omega^{2}_{o})}=j\sqrt{(\omega^{2}_{o}-\lambda^{2})}=j\gamma \end{equation*}
where
\begin{equation*} \gamma = \sqrt{(\omega^{2}_{o}-\lambda^{2})}. \end{equation*}
Hence, from eqn. (3.7.14), we have
\begin{equation*} q= q_{o}-\frac{q_{o}}{2}\left(1+\frac{\lambda}{j\gamma}\right) e^{(-\lambda +j\gamma)t} - \frac{q_{o}}{2}\left(1-\frac{\lambda}{j\gamma}\right) e^{(-\lambda -j\gamma)t} \end{equation*}
\begin{equation*} \text{or,}\quad q = q_{o}-\frac{q_{o}}{2}e^{-\lambda t}\left(e^{j\gamma t} +\frac{\lambda}{j\gamma} e^{j\gamma t}+ e^{-j\gamma t}- \frac{\lambda}{j\gamma} e^{-j\gamma t}\right) \end{equation*}
\begin{equation*} \text{or,}\quad q = q_{o}-q_{o}e^{-\lambda t}\left(\frac{e^{j\gamma t}+e^{-j\gamma t}}{2} + \frac{\lambda}{\gamma} \frac{e^{j\gamma t}-e^{-j\gamma t}}{2j} \right) \end{equation*}
\begin{equation*} \therefore\quad q = q_{o}-q_{o}e^{-\lambda t}\left(\cos\gamma t+ \frac{\lambda}{\gamma} \sin\gamma t\right) \end{equation*}
