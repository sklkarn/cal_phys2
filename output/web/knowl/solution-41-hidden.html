Given: \(E=6\,V; \quad C = 5.00\times10^{-6}\,F;\) \(R = 2.00\times10^{6}\,\Omega; \quad t =6\,s. \)
\begin{equation*} E =\frac{q}{C}+IR; \quad I=\frac{\,dq}{\,dt};\quad V_{r} = IR \end{equation*}
\begin{equation*} E- \frac{q}{C} =\frac{\,dq}{\,dt}R; \end{equation*}
\begin{equation*} or,\quad \frac{E}{R}- \frac{q}{RC} =\frac{\,dq}{\,dt} \end{equation*}
\begin{equation*} or,\quad \frac{EC-q}{RC} =\frac{\,dq}{\,dt}; \end{equation*}
\begin{equation*} or, \quad \frac{\,dq}{EC-q} = \frac{\,dt}{RC} \end{equation*}
\begin{equation*} or, \quad \int\limits_{0}^{Q}\frac{\,dq}{EC-q}=\int\limits_{0}^{t}\frac{\,dt}{RC}; \end{equation*}
\begin{equation*} or, \quad \left.\frac{\ln(EC-q)}{-1}\right]_{0}^{Q} = \frac{t}{RC} \end{equation*}
\begin{equation*} or, \quad \ln(EC-Q)-\ln(EC) =-\frac{t}{RC}; \end{equation*}
\begin{equation*} or, \quad \ln\frac{EC-Q}{EC}=-\frac{t}{RC} \end{equation*}
\begin{equation*} or, \quad EC-Q =ECe^{-\frac{t}{RC}}; \end{equation*}
\begin{equation*} or, \quad Q = EC[1-e^{\frac{-t}{RC}}] \end{equation*}
\begin{equation*} or, \quad Q=Q_{o}[1-e^{\frac{-t}{\tau}}] \end{equation*}
differentiate w.r.t ’t’, we get-
\begin{equation*} I =I_{o}e^{-t/\tau} \end{equation*}
\begin{equation*} \text{let}\quad I_{o}= \frac{E}{R} = 3.0\,\mu A; \quad \tau = RC;\quad EC = Q_{o} [\text{maximum charge}] \end{equation*}
Now,
  1. \(\displaystyle \tau = 10\,s;\quad Q_{o}= 30 \,\mu C;\quad e^{-\frac{t}{\tau}}= 0.606 \)
  2. \(\displaystyle Q=11.81 \,\mu C\)
  3. \(\displaystyle V_{R}= 3.639 \,V\)
  4. \(\displaystyle I = 1.82 \,\mu A\)
