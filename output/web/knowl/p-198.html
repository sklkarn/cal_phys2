  1. \begin{align*} \vec{F}_{elect} \amp =k\frac{q_{1}q_{2}}{r^{2}}\hat{r}\\ \vec{F}_{ext} \amp =-\vec{F}_{elect}; \end{align*}
    \(r =x;\quad \hat{r}=\hat{i};\quad \vec{\,dl}=\,dx(\hat{i})\)
    \begin{align*} W_{ab}\amp =\int\limits_{a}^{b}\vec{F}_{ext}\cdot \vec{\,dl} = -\int\limits_{1}^{2}F_{elect}\,dx \\ \amp =-k\int\limits_{1}^{2}\frac{q_{1}q_{2}}{x^{2}}\,dx =-k \left[-\frac{q_{1}q_{2}}{x}\right]_{1}^{2} \\ \amp = kq_{1}q_{2}\left(\frac{1}{2}-\frac{1}{1}\right) =-0.90 \,J \end{align*}
  2. \begin{equation*} V_{b}=k\frac{q_{1}}{r_{b}}=9.0\times 10^{4} \,V,\quad [r_{b}= 1\,m ] \end{equation*}
  3. \begin{align*} V_{c}\amp =k\frac{q_{1}}{r_{c}}=4.5\times 10^{4} \,V,\quad [r_{c}= 2\,m] \\ V_{bc}\amp =V_{b}-V_{c} = 4.5\times 10^{4} \,V \end{align*}
  4. \begin{align*} V_{d}\amp =k\frac{q_{1}}{r_{d}}=9.0\times 10^{4} \,V,\quad [r_{d}= 1\,m]\\ V_{bd}=V_{b}-V_{d} \amp =0 \end{align*}
  5. \begin{equation*} W_{bd}=q_{2}(V_{d}-V_{b}) = 0 \end{equation*}
  6. \begin{equation*} \int\limits_{b}^{d}\vec{E}\cdot \vec{dl}=-V_{bd} =0 \end{equation*}
    The same result is obtained by taking \(\vec{E}\) is always perpendicular to \(\vec{\,dl}\) along the circular path of integration, so
    \begin{align*} \vec{E}\cdot\vec{dl} \amp = 0 \\ \text{and} \quad \int\limits_{b}^{d}\vec{E}\cdot \vec{dl} \amp =0. \end{align*}
