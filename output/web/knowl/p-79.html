Given: \(E = 100 \,N/C;\quad L = 1 \,m; \quad r = 0.2 \,m. \)
\begin{equation*} \text{Area of the caps}\quad A_{1}=\pi r^{2} = A_{2} =0.1257 \,m^{2} \end{equation*}
\begin{equation*} \text{Area of the cylindrical surface}\quad A_{3} =2 \pi r L =1.257 \,m^{2} \end{equation*}
\begin{equation*} \text{Now,}\quad \Phi_{1} = \int\vec{E}\cdot\vec{dA} = EA_{1} =12.5 \,Nm^{2}/C \end{equation*}
\begin{equation*} \Phi_{2} = \int \vec{E}\cdot \vec{\,dA}= - \int E \,dA = -EA_{2} = -12.5 \,Nm^{2}/C \end{equation*}
\begin{equation*} \Phi_{3} = \vec{E}\cdot \vec{A_{3}} = 0 \end{equation*}
\begin{equation*} \therefore\quad \Phi_{T} =\Phi_{1}+\Phi_{2}+\Phi_{3}=0 \end{equation*}
