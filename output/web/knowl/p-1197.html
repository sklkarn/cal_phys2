\begin{equation*} \text{or,} \quad y = \frac{aA\sin\alpha}{\alpha}\left[\sin(\omega t -kr) + \sin\left(\omega t -kr+kd\sin\theta\right) \right] \end{equation*}
\begin{equation*} \text{or,} \quad y = \frac{2aA\sin\alpha}{\alpha}\left[\sin(\omega t -kr) + \frac{kd\sin\theta}{2}\right] \cos\left(-\frac{kd\sin\theta}{2} \right) \end{equation*}
Let
\begin{equation*} \frac{kd\sin\theta}{2}=\beta, \end{equation*}
and
\begin{equation*} \cos(-\theta) = \cos\theta, \end{equation*}
we have -
\begin{equation} \therefore \quad y = \frac{2aA\sin\alpha}{\alpha} \cos\beta \sin(\omega t -kr+\beta) \tag{5.8.24} \end{equation}
The intensity at point P is therefore given by
\begin{equation} I=4a^{2}A^{2}\left(\frac{\sin^{2}\alpha}{\alpha^{2}}\right)\cos^{2}\beta = 4I_{o}\left(\frac{\sin\alpha}{\alpha}\right)^{2}\cos^{2}\beta \tag{5.8.25} \end{equation}
The intensity at central maximum is \(4I_{o}\) when \(\alpha=0\) and \(\beta=0\text{.}\)
