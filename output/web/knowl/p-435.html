\begin{equation*} \vec{B}=B\cos\theta\hat{i}+B\sin\theta\hat{j}; \end{equation*}
\begin{equation*} \vec{A}=\pi r^{2}\hat{j} \end{equation*}
\begin{equation*} \Phi_{B}=\int \vec{B}\cdot\vec{dA}=\vec{B}\cdot\vec{A}; \end{equation*}
\begin{equation*} \vec{\mu}=NI\vec{A}; \end{equation*}
\begin{equation*} \vec{\tau}=\vec{\mu}\times\vec{B}; \end{equation*}
\begin{equation*} U = -\vec{\mu}\cdot\vec{B} \end{equation*}
