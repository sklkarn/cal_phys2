\begin{equation*} \text{or,}\quad E=\left(R+j\omega L + \frac{1}{j\omega C}\right)I; \end{equation*}
\begin{equation*} \text{or,}\quad E= \left[R+j\omega L - \frac{j}{\omega C}\right]I \end{equation*}
\begin{equation*} \therefore\quad I = \frac{E}{Z}= \frac{E}{\left[R+j\left(\omega L - \frac{1}{\omega C}\right)\right]} \end{equation*}
\begin{equation*} \therefore \quad Z= \left[R+j\left(\omega L - \frac{1}{\omega C}\right)\right] = \left[R-j\left( \frac{1}{\omega C}-\omega L \right)\right] \end{equation*}
Minus sign between \(X_{L}=\omega L\) and \(X_{C}= \frac{1}{\omega C}\) indicates that they are in opposite phase to each other. The circuit becomes inductive, capacitive or purely resistive if \(X_{L} \gt X_{C},\) \(X_{L} \lt X_{C},\) or \(X_{L} = X_{C}.\) Now since impedance \(Z=|Z|e^{j\theta}\text{,}\) we have
\begin{equation*} |Z|= \sqrt{R^{2}+\left(\frac{1}{\omega C}-\omega L \right)^{2}} \end{equation*}
\begin{equation*} \text{and}\quad \theta =-\tan^{-1}\left[\frac{\left(\frac{1}{\omega C}-\omega L\right)}{R} \right] \end{equation*}
\begin{equation*} \therefore \quad Z=\sqrt{R^{2}+\left(\frac{1}{\omega C}-\omega L \right)^{2}} e^{-j\tan^{-1}\left[\frac{\left(\frac{1}{\omega C}-\omega L\right)}{R} \right]} \end{equation*}
\begin{equation*} \text{Hence,}\quad I= \frac{E_{o}e^{j\omega t}}{\sqrt{R^{2}+\left(\frac{1}{\omega C}-\omega L \right)^{2}} e^{-j\tan^{-1}\left[\frac{\left(\frac{1}{\omega C}-\omega L\right)}{R} \right]}} \end{equation*}
\begin{equation*} = \frac{E_{o}e^{j(\omega t+\phi)}} {\sqrt{R^{2}+\left(\frac{1}{\omega C}-\omega L \right)^{2}}} \end{equation*}
where phase angle
\begin{equation*} \phi = -\tan^{-1}\left[\frac{\left(\frac{1}{\omega C}-\omega L\right)}{R} \right] \end{equation*}
\begin{equation*} E_{R}=RI = RI_{o}e^{j(\omega t +\phi)} \end{equation*}
\begin{equation*} E_{L}=j\omega L I = j\omega L I_{o}e^{j(\omega t +\phi)} \end{equation*}
\begin{equation*} E_{C}=-j\frac{1}{\omega C}I = -j \frac{1}{\omega C} I_{o}e^{j(\omega t +\phi)} \end{equation*}
