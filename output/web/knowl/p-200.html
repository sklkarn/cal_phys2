\begin{equation*} \phi = \int\vec{E}\cdot\vec{\,dA} = \frac{q_{enc}}{\epsilon_{o}} \end{equation*}
\begin{equation*} \text{or,}\quad \int E\,dA = E(4\pi r^{2}) = \frac{Q}{\epsilon_{o}} \end{equation*}
where \(r\) is the radius of Gaussian surface and \(q\) is the charge enclosed by it.
\begin{equation*} \therefore\quad E = \frac{Q}{4\pi \epsilon_{o}r^{2}}=\frac{kQ}{r^{2}} \end{equation*}
The potential difference between the shells is then,
\begin{equation*} V=V_{b}-V_{a} = -\int\limits_{b}^{a}\vec{E}\cdot\vec{\,dr} = -\int\limits_{b}^{a}E\,dr \end{equation*}
\begin{equation*} or, \quad V = -\int\limits_{b}^{a}\frac{kQ}{r^{2}}\,dr = kQ\left(\frac{1}{r}\right)_{b}^{a}= kQ\left(\frac{1}{a}-\frac{1}{b}\right) \end{equation*}
\begin{equation*} \text{or,}\quad \frac{V}{Q} = k\left(\frac{b-a}{ab}\right) \end{equation*}
Hence the capacitance of the spherical capacitor is given by
\begin{equation*} C=\frac{Q}{V} = \frac{1}{k}\left(\frac{ab}{b-a}\right)= 4\pi\epsilon_{o}\left(\frac{ab}{b-a}\right) \end{equation*}
If \(b \rightarrow\infty\text{,}\) then
\begin{equation*} C=4\pi\epsilon_{o}\left(\frac{a}{1-\frac{a}{b}}\right)= 4\pi\epsilon_{o} a \end{equation*}
Which is the capacitance of an isolated spherical conductor of radius \(a\text{.}\)
