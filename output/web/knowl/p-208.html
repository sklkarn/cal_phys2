Let us consider two capacitors of capacitance \(C_{1}, \quad C_{2}\) and have potentials \(V_{1}\) and \(V_{2},\) respectively. Hence total charge on the capacitors is
\begin{equation} q=q_{1}+q_{2} \tag{1.7.17} \end{equation}
When connected charge from one capacitor transfers to another till they obtain common potential \(V, \) i.e.
\begin{equation} q=q'_{1} + q'_{2} \tag{1.7.18} \end{equation}
\begin{equation*} \text{or,} \quad q_{1}+q_{2} = q'_{1} + q'_{2} \end{equation*}
\begin{equation*} C_{1}V_{1}+C_{2}V_{2} = C_{1}V + C_{2}V =V(C_{1} + C_{2}) \end{equation*}
\begin{equation} \text{or,} \quad V= \frac{C_{1}V_{1}+C_{2}V_{2}}{C_{1} + C_{2}} \tag{1.7.19} \end{equation}
Electrical energy of the capacitors before and after connection
\begin{equation*} U=\frac{1}{2} C_{1}V^{2}_{1}+\frac{1}{2} C_{2}V^{2}_{2} \end{equation*}
\begin{equation*} \text{and} \quad U' =\frac{1}{2} C_{1}V^{2}+\frac{1}{2} C_{2}V^{2} = \frac{1}{2} \left(C_{1}+C_{1}\right)V^{2} \end{equation*}
Hence energy loss after the connection is given by
\begin{equation*} \Delta U = U-U' =\frac{1}{2} C_{1}V^{2}_{1}+\frac{1}{2} C_{2}V^{2}_{2} -\frac{1}{2} \left(C_{1}+C_{2}\right)V^{2} \end{equation*}
\begin{equation*} =\frac{1}{2} C_{1}V^{2}_{1}+\frac{1}{2} C_{2}V^{2}_{2} -\frac{1}{2} \left(C_{1}+C_{2}\right)\left[ \frac{C_{1}V_{1}+C_{2}V_{2}}{C_{1} + C_{2}}\right]^{2} \end{equation*}
\begin{equation*} =\frac{1}{2} C_{1}V^{2}_{1}+\frac{1}{2} C_{2}V^{2}_{2} -\frac{1}{2} \left(\frac{1}{C_{1}+C_{2}}\right)\left[ C_{1}V_{1}+C_{2}V_{2}\right]^{2} \end{equation*}
\begin{equation*} =\frac{1}{2} \left(\frac{1}{C_{1}+C_{2}}\right)\left[C^{2}_{1}V^{2}_{1}+ C_{1}C_{2}V^{2}_{1}+C_{1}C_{2}V^{2}_{2}+C^{2}_{2}V^{2}_{2}-C^{2}_{1}V^{2}_{1}-C^{2}_{2}V^{2}_{2}-2C_{1}C_{2}V_{1}V_{2}\right] \end{equation*}
\begin{equation*} =\frac{1}{2} \left(\frac{1}{C_{1}+C_{2}}\right)\left[C_{1}C_{2}\left(V^{2}_{1}+V^{2}_{2}-2V_{1}V_{2}\right)\right] \end{equation*}
\begin{equation*} = \frac{C_{1}C_{2}}{2\left(C_{1}+C_{2}\right)}\left(V_{1}-V_{2}\right)^{2} \end{equation*}
Since \(\left(V_{1}-V_{2}\right)^{2} \gt 0\text{,}\) there must be energy loss when connecting two capacitors.
