\begin{equation*} \,dB= \frac{\mu_{o}IN a^{2}}{2(a^{2}+x^{2})^{3/2}} \end{equation*}
from eqn. (3.4.2)
\begin{equation*} \therefore\quad \,dB= \frac{\mu_{o}I(n\,dl) a^{2}}{2(a^{2}+x^{2})^{3/2}} \end{equation*}
From \(\triangle CDE,\) \(\sin\phi=\frac{CE}{CD}.\)
\begin{equation*} \text{or,}\quad CD= \,dl=\frac{CE}{\sin\phi} = \frac{r\,d\phi}{\sin\phi} = r\,d\phi\csc\phi \end{equation*}
\begin{equation*} \text{also,}\quad r^{2} = a^{2}+x^{2} \end{equation*}
\begin{equation*} \text{and}\quad \sin\phi= \frac{a}{r} \end{equation*}
\begin{equation*} \therefore\quad \,dB =\frac{\mu_{o}(nr\,d\phi\csc\phi)I a^{2}}{2r^{3}} = \frac{\mu_{o}nI}{2}\frac{a^{2}}{r^{2}}\csc\phi\,d\phi \end{equation*}
\begin{equation*} =\frac{\mu_{o}nI}{2}\sin^{2}\phi\csc\phi\,d\phi =\frac{\mu_{o}nI}{2}\sin\phi\,d\phi \end{equation*}
