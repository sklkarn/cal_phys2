  1. For the first secondary maximum \(\alpha=\frac{3\pi}{2}\text{,}\)
    \begin{equation*} \therefore\quad I =I_{o}\left(\frac{\sin\alpha}{\alpha}\right)^{2} \end{equation*}
    \begin{equation*} = I_{o} \left(\frac{-1}{\frac{3\pi}{2}}\right)^{2} =\frac{4I_{o}}{9\pi^{2}} \end{equation*}
  2. For the second secondary maximum \(\alpha=\frac{5\pi}{2}\text{,}\)
    \begin{equation*} \therefore\quad I =I_{o}\left(\frac{\sin\alpha}{\alpha}\right)^{2} \end{equation*}
    \begin{equation*} = I_{o} \left(\frac{1}{\frac{5\pi}{2}}\right)^{2} =\frac{4I_{o}}{25\pi^{2}} \end{equation*}
