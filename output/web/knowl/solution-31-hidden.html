1. \begin{equation*} \sigma_{a}=\frac{Q}{A_{a}}; \end{equation*}
    \begin{equation*} \text{or,}\quad A_{a}= \pi r^{2}_{a}; \end{equation*}
    \begin{equation*} \therefore\quad \sigma_{a}=-\frac{Q}{4\pi r^{2}_{a}} \end{equation*}
  2. \begin{equation*} \sigma_{b}=\frac{Q}{A_{b}}; \end{equation*}
    \begin{equation*} \text{or,}\quad A_{b}= \pi r^{2}_{b} \end{equation*}
    \begin{equation*} \therefore\quad \sigma_{b}=-\frac{Q}{4\pi r^{2}_{b}} \end{equation*}
  3. \begin{equation*} \text{spherical charge} \quad E=-k\frac{Q}{ r^{2}} \end{equation*}
  4. \begin{equation*} V_{a}=-k \frac{Q}{r_{a}}; \end{equation*}
    \begin{equation*} V_{b}=-k \frac{Q}{r_{b}}; \end{equation*}
    \begin{equation*} V=V_{ba} =V_{b}-V_{a} \end{equation*}
    \begin{equation*} V= kQ\left[\frac{1}{r_{a}}-\frac{1}{r_{b}}\right] \end{equation*}
  5. \begin{equation*} C = \frac{Q}{V} = 4\pi\epsilon_{o}\frac{r_{a}r_{b}}{r_{b}-r_{a}} \end{equation*}
  6. Replace \(Q\) in the above equations with \(-Q.\) The sign on every answer would change except for the last. The capacitance would be the same.
