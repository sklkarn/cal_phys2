\begin{equation*} \text{Now}\quad L=\frac{\lambda_{1}}{4} \qquad \Rightarrow \lambda_{1} = 4L \end{equation*}
\begin{equation*} \text{or,}\quad f_{1} =\frac{v}{\lambda_{1}}= \frac{v}{4L}=f_{o} \quad [\text{I Harmonics or fundamental frequency.}] \end{equation*}
\begin{equation*} \text{again}\quad L=\frac{\lambda_{2}}{2} +\frac{\lambda_{2}}{4} = 3\left(\frac{\lambda_{2}}{4}\right) \end{equation*}
\begin{equation*} \Rightarrow \lambda_{2} = \frac{4L}{3} \end{equation*}
\begin{equation*} \text{and} \quad f_{2} = \frac{v}{\lambda_{2}} =3\left(\frac{v}{4L}\right)=3f_{o} \quad [\text{III Harmonics or I overtone.}] \end{equation*}
\begin{equation*} \text{similarly}\quad L=\frac{\lambda_{3}}{4} +\frac{\lambda_{3}}{2}+\frac{\lambda_{3}}{2} = 5\left(\frac{\lambda_{3}}{4}\right) \end{equation*}
\begin{equation*} \Rightarrow \lambda_{3} = \frac{4L}{5} \end{equation*}
\begin{equation*} \text{and} \quad f_{3} = \frac{v}{\lambda_{3}} =5\left(\frac{v}{4L}\right)=5f_{o} \end{equation*}
\begin{equation*} \therefore\quad f_{3} = 5f_{o} \quad [\text{V Harmonics or II overtone.}] \end{equation*}
