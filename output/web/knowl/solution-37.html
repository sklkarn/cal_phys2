Solution 2.5.1.1.

(Find equivalent resistance and work backward to get voltage and current.)
Given: \(r= 1\,\Omega;\quad R_{1} =1 \,\Omega; \quad R_{2}= 2\Omega;\) \(R_{3}= 6\,\Omega;\quad R_{4}= 3\,\Omega;\quad E = 12\,V\)
Since \(R_{1}\) and \(R_{2}\) are in series, we have
\begin{equation*} R_{12} = R_{1}+R_{2} = 3 \,\Omega; \end{equation*}
Since \(R_{12}\) and \(R_{3}\) are in parallel, we have
\begin{equation*} \frac{1}{R_{123}}= \frac{1}{R_{12}} + \frac{1}{R_{3}}; \end{equation*}
\begin{equation*} \therefore R_{123} = 2 \,\Omega \end{equation*}
Since \(r, \quad R_{123},\) and \(R_{4}\) are in series, we have
\begin{equation*} R_{1234}= r + R_{123} + R_{4} = 6 \,\Omega \end{equation*}
Now, from figure below, we have
\begin{equation*} \left[ V = I R \right]; \end{equation*}
\begin{equation*} or, \quad I = \frac{V}{R} = \frac{E}{R_{1234}} = 12/6 = 2\,A \end{equation*}
Moving backward: from figure
\begin{equation*} V_{o} = I r =2\,A\cdot1\,\Omega = 2\,V; \end{equation*}
\begin{equation*} \text{ also,} \quad V_{4} = I R_{4}= 2\,A\cdot3\,\Omega = 6\,V \end{equation*}
\begin{equation*} \text{and} \quad V_{123} = I R_{123}= 2\,A\cdot2\,\Omega = 4\,V \end{equation*}
from figure
\begin{equation*} \because \quad V_{12} = I_{12} R_{12}; \end{equation*}
\begin{equation*} \therefore I_{12} = \frac{V_{12}}{R_{12}} = \frac{V_{123}}{R_{12}} = 1.33 \,A \end{equation*}
\begin{equation*} \text{also,} \quad \because \quad V_{3} = I_{3} R_{3}; \end{equation*}
\begin{equation*} \therefore I_{3} = \frac{V_{3}}{R_{3}} = \frac{V_{123}}{R_{3}} = 0.67 \,A \end{equation*}
from figure
\begin{equation*} \because \quad V_{1} = I_{1} R_{1} = I_{12} R_{1} = 1.33\,A\cdot 1\,\Omega = 1.33 \,V \end{equation*}
\begin{equation*} \text{also,}\quad \because \quad V_{2} = I_{12} R_{2}=1.33\,A\cdot 2\,\Omega = 2.67\, V \end{equation*}
  1. \begin{equation*} V_{o}=2\,V; \end{equation*}
    \begin{equation*} I_{o}=2\,A; \end{equation*}
    \begin{equation*} V_{1}=1.33\,V; \end{equation*}
    \begin{equation*} I_{1}=1.33\,A; \end{equation*}
    \begin{equation*} V_{2}=2.67\,V; \end{equation*}
    \begin{equation*} I_{2}=1.33\,A; \end{equation*}
    \begin{equation*} V_{4}=6\,V; \end{equation*}
    \begin{equation*} I_{4}=2\,A; \end{equation*}
    \begin{equation*} V_{3}=4\,V; \end{equation*}
    \begin{equation*} I_{3}=0.67\,A \end{equation*}
  2. \begin{equation*} V_{ab}=E - V_{o}= 10\,V \end{equation*}
  3. \begin{equation*} V_{c}= V_{cb}=V_{2}+V_{4} = 8.67\,V \end{equation*}
