\begin{equation*} I = \frac{L\cos\theta}{r^{2}} \end{equation*}
At center point P on the table,
\begin{equation*} I_{p} = \frac{L}{r^{2}} = \frac{100}{6^{2}} \end{equation*}
\([\because \cos\theta =0]\)
\begin{equation*} \therefore \quad I_{p} = 2.77 \,Lumen/m^{2} \end{equation*}
At the edge point Q of the table,
\begin{equation*} I_{q} = \frac{L\cos\theta}{R^{2}} = \frac{100\times \frac{6}{10}}{10^{2}} \end{equation*}
\([\because R = \sqrt{x^{2}+y^{2}} =\sqrt{8^{2}+6^{2}} =10; \quad \cos\theta =\frac{6}{10}]\)
\begin{equation*} \therefore \quad I_{q} = 0.6 \,Lumen/m^{2} \end{equation*}
