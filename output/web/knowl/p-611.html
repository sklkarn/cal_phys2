Let emf \(E=E_{o}\sin\omega t \) be applied to a circuit containing a resistor of resistance R, an inductor of inductance L, and a capacitor of capacitance C in parallel as shown in FigureÂ 3.9.12.(a). Hence,
\begin{equation*} E(t) = V_{R}=V_{L}=V_{C} \end{equation*}
Now,
\begin{equation*} I_{R}(t)=\frac{E(t)}{R} = \frac{E_{o}}{R} \sin\omega t = I_{Ro} \sin\omega t \end{equation*}
\begin{equation*} V_{L}(t)=E(t)=E_{o}\sin\omega t = L\frac{\,dI_{L}}{\,dt} \end{equation*}
\begin{equation*} \therefore I_{L}(t)= \int_{0}^{t} \frac{E_{o}}{L}\sin\omega t'\,dt' \end{equation*}
\begin{equation*} =-\frac{E_{o}}{\omega L}\cos\omega t = \frac{E_{o}}{X_{L}}\sin\left(\omega t-\frac{\pi}{2}\right) \end{equation*}
\begin{equation*} =I_{Lo} \sin\left(\omega t-\frac{\pi}{2}\right) \end{equation*}
\begin{equation*} V_{C}(t)=E(t)=E_{o}\sin\omega t = \frac{Q(t)}{C} \end{equation*}
\begin{equation*} \therefore I_{C}(t)= \frac{\,dQ(t)}{\,dt} \end{equation*}
\begin{equation*} =\omega CE_{o}\cos\omega t = X_{C}E_{o}\sin\left(\omega t+\frac{\pi}{2}\right) \end{equation*}
\begin{equation*} =I_{Co} \sin\left(\omega t+\frac{\pi}{2}\right) \end{equation*}
