\begin{equation*} \sin\theta = \frac{y}{A} \end{equation*}
\begin{equation*} \text{or,}\quad y=A \sin\theta \end{equation*}
\begin{equation*} \therefore \quad\,dy = A\cos\theta\,d\theta \end{equation*}
\begin{equation*} \text{or,}\quad \mathscr{I} = \int\frac{A\cos\theta\,d\theta}{A\cos\theta} = \int\,d\theta = \theta = \sin^{-1}\frac{y}{A} \end{equation*}
Hence from eqn. (4.1.2), we have -
\begin{equation*} \left.\sin^{-1}\frac{y}{A}\right\vert_{y_{o}}^{y}=\omega t \end{equation*}
\begin{equation*} \text{or,}\quad \sin^{-1}\frac{y}{A}-\sin^{-1}\frac{y_{o}}{A}=\omega t \end{equation*}
\begin{equation*} \text{or,}\quad \sin^{-1}\frac{y}{A} =\omega t+\sin^{-1}\frac{y_{o}}{A} = \omega t+\phi_{o} \end{equation*}
\begin{equation} \text{or,}\quad y = A\sin \left(\omega t+\phi_{o}\right) \tag{4.1.3} \end{equation}
