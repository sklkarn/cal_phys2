\begin{equation*} \vec{F}=I\vec{l}\times\vec{B} \end{equation*}
\begin{equation*} (ab) \quad \vec{l}=a\hat{k}; \end{equation*}
\begin{equation*} (bc) \quad \vec{l}=a\hat{i}-a\hat{k}; \end{equation*}
\begin{equation*} (cd) \quad \vec{l}=-a\hat{j}; \end{equation*}
\begin{equation*} (de) \quad \vec{l}=a\hat{j}+a\hat{k}; \end{equation*}
\begin{equation*} (ef) \quad \vec{l}=-a\hat{i}-a\hat{j}- a\hat{k} \end{equation*}
  1. \begin{equation*} \text{Along (ab)} \quad \vec{F} = 1.00\,N \hat{j}; \end{equation*}
  2. \begin{equation*} \text{Along (bc)} \quad \vec{F} = 1.00\,N [-\hat{j}] \end{equation*}
  3. \begin{equation*} \text{Along (cd)} \quad \vec{F} = 1.00\,N \hat{k}; \end{equation*}
  4. \begin{equation*} \text{Along (de)} \quad \vec{F} = 1.41 \,N \left[\frac{\hat{j}-\hat{k}}{\sqrt{2}}\right] \end{equation*}
  5. \begin{equation*} \text{Along (ef)}\quad \vec{F} = 1.41 \,N \left[\frac{-\hat{j}+\hat{k}}{\sqrt{2}}\right] \end{equation*}
