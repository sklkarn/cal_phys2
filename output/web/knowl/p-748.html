\begin{equation*} \text{Here}\quad L=\frac{\lambda_{1}}{2} \qquad \Rightarrow \lambda_{1} = 2L; \end{equation*}
\begin{equation*} \text{Also}\quad v=f_{1}\lambda_{1} \qquad \Rightarrow f_{1} = \frac{v}{\lambda_{1}} \end{equation*}
where \(\lambda\text{,}\) f and v are wavelength, frequency, and velocity of a progressive wave.
\begin{equation*} \text{or,}\quad f_{1} = \frac{v}{2L}=f_{o} \end{equation*}
This is I Harmonics or fundamental frequency.
\begin{equation*} \text{again}\quad L=\frac{\lambda_{2}}{2} +\frac{\lambda_{2}}{2} = \lambda_{2} \end{equation*}
\begin{equation*} \Rightarrow \lambda_{2} = L \end{equation*}
\begin{equation*} \text{and} \quad f_{2} = \frac{v}{\lambda_{2}} =\frac{v}{L} \end{equation*}
\begin{equation*} \therefore\quad f_{2} = \frac{v}{L}=2\left(\frac{v}{2L}\right)=2f_{o} \end{equation*}
This is II Harmonics or I overtone.
\begin{equation*} \text{similarly}\quad L=\frac{\lambda_{3}}{2} +\frac{\lambda_{3}}{2}+\frac{\lambda_{3}}{2} = 3\frac{\lambda_{3}}{2} \end{equation*}
\begin{equation*} \Rightarrow \lambda_{3} = \frac{2L}{3} \end{equation*}
\begin{equation*} \text{and} \quad f_{3} = \frac{v}{\lambda_{3}} =3\left(\frac{v}{2L}\right)=3f_{o} \end{equation*}
\begin{equation*} \therefore\quad f_{3} = 3f_{o} \end{equation*}
This is III Harmonics or II overtone.
