\begin{equation*} \text{or,} \quad \rho = r-z\sin\theta \end{equation*}
\begin{equation} \therefore \quad y=\int\limits_{-\frac{a}{2}}^{+\frac{a}{2}} A\sin\left[\omega t - k(r-z\sin\theta)\right]\,dz \tag{5.8.14} \end{equation}
\begin{equation*} \text{or,} \quad y=\left.\frac{-A\cos\left[\omega t - k(r-z\sin\theta)\right]}{k\sin\theta}\right]_{-\frac{a}{2}}^{+\frac{a}{2}} \end{equation*}
\begin{equation*} \text{or,} \quad y =-\frac{A}{k\sin\theta}\left[\cos\left\{\omega t -k\left(r-\frac{a}{2}\sin\theta\right)\right\}-\cos\left\{\omega t -k\left(r+\frac{a}{2}\sin\theta\right)\right\}\right] \end{equation*}
\begin{equation*} \text{or,} \quad y =-\frac{A}{k\sin\theta}\left[\cos\left\{(\omega t -kr)+\frac{ka}{2}\sin\theta \right\}-\cos\left\{(\omega t -kr)-\frac{ka}{2}\sin\theta \right\}\right] \end{equation*}
\begin{equation} \therefore \quad y = \frac{2A}{k\sin\theta}\sin\left(\frac{ka}{2}\sin\theta\right) \sin(\omega t-kr) \tag{5.8.15} \end{equation}
\begin{equation*} \text{Let}\quad \frac{ka}{2}\sin\theta =\alpha \end{equation*}
\begin{equation} \therefore \quad y = Aa\frac{\sin\alpha}{\alpha} \sin(\omega t-kr) \tag{5.8.16} \end{equation}
Therefore the amplitude at P is \(\left(Aa\frac{\sin\alpha}{\alpha}\right)\) and thus intensity at P is
\begin{equation*} I=\left(Aa\frac{\sin\alpha}{\alpha}\right)^{2} = A^{2}a^{2}\left( \frac{\sin^{2}\alpha}{\alpha^{2}}\right) \end{equation*}
\begin{equation} \therefore\quad I = I_{o} \left(\frac{\sin^{2}\alpha}{\alpha^{2}}\right) \tag{5.8.17} \end{equation}
\(\because\quad \lim\limits_{\alpha \to 0} \frac{\sin\alpha}{\alpha}=1\) we have
\begin{equation*} I=I_{o}\quad [\textbf{central maximum},\quad as \quad \alpha \to 0, \theta \to 0.] \end{equation*}
For secondary maxima, from eqn. (5.8.11),
\begin{equation*} \sin\theta_{n} = \frac{(2n+1)\lambda}{2a} \end{equation*}
\begin{equation} \therefore\quad \alpha = \frac{ka}{2}\frac{(2n+1)\lambda}{2a} = \frac{(2n+1)\pi}{2} \quad [k=\frac{2\pi}{\lambda}] \tag{5.8.18} \end{equation}
Substituting \(n=1,2,3,\cdots\text{,}\) we have
\begin{equation*} \alpha = \frac{3\pi}{2}, \frac{5\pi}{2},\frac{7\pi}{2}, \cdots \end{equation*}
