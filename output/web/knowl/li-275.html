Item 3.e.

\begin{equation*} \sin(-\theta) = -\sin\theta; \end{equation*}
\begin{equation*} \cos(-\theta) = \cos\theta; \end{equation*}
\begin{equation*} \sin(\theta\pm \frac{\pi}{2}) = \pm \cos\theta; \end{equation*}
\begin{equation*} \cos(\theta\pm \frac{\pi}{2}) = \mp \sin\theta. \end{equation*}
