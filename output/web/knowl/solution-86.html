Solution 5.5.10.1.

  1. \begin{equation*} \frac{1}{f} = (n-1)\left(\frac{1}{R_{1}}-\frac{1}{R_{2}}\right) \end{equation*}
    \begin{equation*} or,\quad \frac{1}{f} = (1.5-1)\left(\frac{1}{-10}-\frac{1}{10}\right) \end{equation*}
    \begin{equation*} \Rightarrow f=-10\,cm \quad\text{diverging.} \end{equation*}
  2. \begin{equation*} \frac{1}{s}+\frac{1}{s'}=\frac{1}{f} \end{equation*}
    \begin{equation*} or,\quad \frac{1}{50}+ \frac{n'}{s'}= \frac{1}{-10} \end{equation*}
    \begin{equation*} \Rightarrow s' = -12.5 \,cm \quad \text{virtual.} \end{equation*}
  3. \begin{equation*} m=\frac{y'}{y}=-\frac{s'}{s} \end{equation*}
    \begin{equation*} or,\quad \frac{y'}{2} =-\frac{-12.5}{50} \end{equation*}
    \begin{equation*} \Rightarrow y' =0.5\, cm; \quad \text{reduced erect.} \end{equation*}
