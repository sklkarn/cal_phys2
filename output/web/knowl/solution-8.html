Solution 1.3.8.1.

Given: \(q = 2 e; \quad e = 1.60\times 10^{-19} \,C; \quad d = 4.00\times 10^{-11} \,m; \) \(\theta = 60^{o}; \quad \phi = 45^{o}; \quad E= 100 (N/C).\)
  1. \begin{equation*} \vec{p}=\vec{p_{1}}+\vec{p_{2}} \end{equation*}
    \begin{equation*} \vec{p_{1}} = e d(\cos\theta\hat{i}+\sin\theta \hat{j}); \end{equation*}
    \begin{equation*} \vec{p_{2}} = e d(\cos\theta\hat{i}-\sin\theta \hat{j}); \end{equation*}
    \begin{equation*} \vec{p}=2e d\cos\theta \hat{i}=6.40\times 10^{-30}\,Cm \hat{i} \end{equation*}
  2. \begin{equation*} F = q\vec{E} \end{equation*}
    \begin{equation*} \vec{E} = E(\cos\phi\hat{i}+\sin\phi \hat{j}); \end{equation*}
    \begin{equation*} \vec{F} = 2.26\times 10^{-17}\,N(\hat{i}+ \hat{j}) \end{equation*}
  3. \begin{equation*} \tau=\vec{p}\times\vec{E} \end{equation*}
    \begin{equation*} \vec{p}=p\hat{i}; \end{equation*}
    \begin{equation*} \vec{E}=E\cos\phi\hat{i}+E\sin\phi \hat{j}; \end{equation*}
    \begin{equation*} \tau= \begin{bmatrix} \hat{i} & \hat{j} & \hat{k}\\ p & 0 & 0 \\ E \cos \phi & E\sin \phi & 0 \end{bmatrix} = pE\sin\phi \hat{k}=4.53\times 10^{-28} \,Nm\hat{k} \end{equation*}
  4. \begin{equation*} U = -\vec{p}\cdot\vec{E} = -pE = -4.53\times 10^{-28} \,J \end{equation*}
